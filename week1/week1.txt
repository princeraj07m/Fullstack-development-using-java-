SESSION 3 – SQL

Display all records

SELECT * FROM Student;
SELECT * FROM Course;

Aggregate functions

SELECT COUNT() AS total_students FROM Student;
SELECT department, COUNT() AS dept_count FROM Student GROUP BY department;

Sort by VTU number

SELECT * FROM Student ORDER BY vtu_number ASC;
SELECT * FROM Student ORDER BY vtu_number DESC;

Display students from CSE department

SELECT * FROM Student WHERE department='CSE';

Join VTU number with course and faculty

SELECT s.vtu_number, s.name, c.course_name, c.faculty_email
FROM Student s
JOIN Course c ON s.vtu_number = c.student_id;

SESSION 4 – SQL

Count students per course

SELECT course_name, COUNT(student_id) AS total_students
FROM Course
GROUP BY course_name;

Insert faculty courses into new table

CREATE TABLE Faculty_Course_Assignments AS
SELECT faculty_id, course_name
FROM Course;

Select top 5 students from CSE and update phone with country code

SELECT * FROM Student
WHERE department='CSE'
LIMIT 5;

UPDATE Student
SET phone = CONCAT('+91', phone)
WHERE department='CSE'
LIMIT 5;

DELIMITER ;

User-defined function returning student count by department

DELIMITER $$

CREATE FUNCTION get_department_student_count(dept VARCHAR(50))
RETURNS INT
DETERMINISTIC
BEGIN
DECLARE total INT;
SELECT COUNT(*) INTO total
FROM Student
WHERE department = dept;
RETURN total;
END$$

DELIMITER ;
